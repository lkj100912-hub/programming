target_names = {1: '선생님/상사', 2: '친한 친구'}

# 기능별 함수 설계 1: 사용자 입력 및 예외 처리 담당
# 사용자가 이상한 값을 입력해서 프로그램이 강제 종료되는 것을 막아주는 방패 역할
def get_user_input(prompt_msg, max_choice):
    while True:
        try:
            choice = int(input(prompt_msg))
            if choice < 1 or choice > max_choice:
                print(f"\n[오류] 1부터 {max_choice} 사이의 번호를 입력해주세요.\n")
                continue
            return choice
        except ValueError:
            print("[오류] 문자가 아닌 숫자로 입력해주세요!\n")

# 기능별 함수 설계 2: 변명 생성 로직 담당
# 데이터를 바탕으로 실제 변명을 만들어냄
def generate_excuse(situation_code, target_code):
    # 앞서 설계한 중첩 딕셔너리에서 사용자가 선택한 조건에 맞는 리스트를 꺼내옴
    excuse_list = system_data[situation_code][target_code]
    
    # 파이썬 random 모듈을 사용해 리스트 안에 있는 여러 변명 중 딱 하나를 무작위로 추출
    selected_excuse = random.choice(excuse_list)
    
    situation_str = system_data[situation_code]['name']
    target_str = target_names[target_code]
    
    print("\n" + "="*45)
    print(f"[{situation_str}] 상황에서 [{target_str}]에게 할 변명:")
    print("-" * 45)
    print(f"👉 \"{selected_excuse}\"")
    print("="*45 + "\n")


# 메인 알고리즘 실행부
# 위에서 만든 함수들을 조립해서 프로그램이 실제로 돌아가게 만드는 무한 루프
print('=' * 45)
print("          위기탈출 맞춤형 변명 생성기          ")
print('=' * 45)

while True: # 사용자가 4번을 누르기 전까지 전체 프로그램을 무한 반복시킴
    print("\n어떤 위기 상황이신가요?")
    print("1. 지각   2. 약속 취소/과제 미제출  3. 엄청난 잘못을 일으킴 4. 프로그램 종료")
    sit_choice = get_user_input("상황 번호를 선택하세요: ", 4)
    
    # 4번을 입력하면 break 명령어를 만나 무한 루프가 깨지면서 안전하게 종료됨
    if sit_choice == 4:
        print("\n변명 생성기를 종료합니다. 위기를 잘 극복하시길 바랍니다!")
        break
        
    print("\n누구에게 변명해야 하나요?")
    print("1. 선생님/상사   2. 친구")
    tar_choice = get_user_input("대상 번호를 선택하세요: ", 2)
    
    # 앞서 입력받은 상황 번호와 대상 번호를 변명 생성 함수에 던져주어 최종 출력
    generate_excuse(sit_choice, tar_choice)
